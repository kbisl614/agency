# Phase 1: HVAC AI Operations — Demo Ready (Emergency Lead Capture Only)

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a working end-to-end demo where a test lead submits via SMS/form, Claude qualifies it, and the response is delivered back to a real phone, with all actions logged to Supabase and visible in a dashboard.

**Architecture:** 
- **Frontend:** Next.js dashboard polling Supabase every 10s (Mission Control)
- **Backend:** n8n with Claude API integration for lead qualification
- **Orchestration:** n8n workflow chaining SignalWire → n8n → Supabase → SignalWire SMS
- **Data:** Supabase (MVP, 3 tables: Leads, Actions, Waitlist)
- **Messaging:** SignalWire for SMS

**Tech Stack:** n8n, Claude API (claude-sonnet-4-20250514), Supabase, SignalWire, n8n, Next.js, Python 3.11+

**What's NOT in Phase 1:**
- Jobber integration (comes in Phase 2)
- Cancellation fill (comes in Phase 2)
- Remaining Layer 1 agents (come in Phase 2)
- Layer 2/3 agents (future phases)

```python
import json
from anthropic import Anthropic
from config import settings

client = Anthropic()

def qualify_lead(customer_name: str, phone: str, message: str, service_type: str = "emergency") -> dict:
    """
    Qualify incoming lead using Claude API.
    Returns structured JSON with urgency_score, confidence, and response_text.
    """
    
    system_prompt = """You are an AI dispatcher for an HVAC business. You receive emergency calls/messages and must qualify them.

Return ONLY valid JSON. No preamble. No explanation.

Schema:
{
  "action": "send_sms" | "escalate",
  "urgency_score": 1-10,
  "service_type": "emergency" | "maintenance" | "seasonal",
  "response_text": "human-sounding SMS response to customer",
  "confidence": 0.0-1.0
}

Rules:
- urgency_score 8-10: AC/heat down in summer/winter
- urgency_score 5-7: partial service, comfort issue
- urgency_score 1-4: routine maintenance, quote request
- confidence >0.85: auto-send SMS
- confidence <0.85: escalate to human
- response_text must be friendly, brief, and confirm next steps
"""

    user_message = f"""
Customer: {customer_name}
Phone: {phone}
Message: {message}
Service Type: {service_type}

Qualify this lead and respond appropriately.
"""

    response = client.messages.create(
        model=settings.CLAUDE_MODEL,
        max_tokens=1000,
        system=system_prompt,
        messages=[
            {"role": "user", "content": user_message}
        ]
    )
    
    # Parse Claude's JSON response
    response_text = response.content[0].text
    result = json.loads(response_text)
    
    return result

def test_claude():
    """Quick test to verify Claude integration works."""
    result = qualify_lead(
        customer_name="John Doe",
        phone="+1234567890",
        message="Our AC went out and it's 95 degrees",
        service_type="emergency"
    )
    print(json.dumps(result, indent=2))
    return result
```

- [ ] **Step 1: Create airtable_service.py**

```python
import json
from datetime import datetime
from pyairtable import Api
from config import settings

api = Api(settings.AIRTABLE_API_KEY)
base = api.base(settings.AIRTABLE_BASE_ID)

def create_lead(customer_name: str, phone: str, message: str, service_type: str) -> str:
    """Create lead record in Supabase. Returns lead_id."""
    leads_table = base.table("Leads")
    
    record = leads_table.create({
        "customer_name": customer_name,
        "phone": phone,
        "message": message,
        "service_type": service_type,
        "status": "new",
        "created_at": datetime.now().isoformat()
    })
    
    return record["id"]

def get_lead(lead_id: str) -> dict:
    """Get lead by ID."""
    leads_table = base.table("Leads")
    return leads_table.get(lead_id)

def update_lead_status(lead_id: str, status: str):
    """Update lead status (new → qualified → dispatched)."""
    leads_table = base.table("Leads")
    leads_table.update(lead_id, {"status": status})

def log_action(lead_id: str, action_type: str, description: str, agent_name: str = "system"):
    """Log action to audit trail."""
    actions_table = base.table("Actions")
    
    actions_table.create({
        "lead_id": [lead_id],  # Link field needs array
        "action_type": action_type,
        "description": description,
        "agent_name": agent_name,
        "timestamp": datetime.now().isoformat()
    })

def get_all_actions(limit: int = 50) -> list:
    """Get recent actions for dashboard (most recent first)."""
    actions_table = base.table("Actions")
    all_actions = actions_table.all()
    return sorted(all_actions, key=lambda x: x["fields"].get("timestamp", ""), reverse=True)[:limit]
```

from twilio.rest import Client
from config import settings

client = Client(settings.TWILIO_ACCOUNT_SID, settings.TWILIO_AUTH_TOKEN)

def send_sms(to_phone: str, message: str) -> str:
    """Send SMS via SignalWire. Returns message SID."""
    try:
        message_obj = client.messages.create(
            body=message,
            from_=settings.TWILIO_PHONE_NUMBER,
            to=to_phone
        )
        return message_obj.sid
    except Exception as e:
        print(f"Error sending SMS: {e}")
        raise

def test_twilio(to_phone: str):
    """Send test SMS."""
    message_sid = send_sms(to_phone, "Test from HVAC AI: System online ✓")
    print(f"Message sent with SID: {message_sid}")

```python
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from services.claude_service import qualify_lead
from services.airtable_service import create_lead, update_lead_status, log_action
from services.twilio_service import send_sms

router = APIRouter(prefix="/leads", tags=["leads"])

class LeadRequest(BaseModel):
    customer_name: str
    phone: str
    message: str
    service_type: str = "emergency"

class LeadResponse(BaseModel):
    lead_id: str
    urgency_score: int
    action: str
    response_text: str
    sms_sent: bool

@router.post("/qualify", response_model=LeadResponse)
async def qualify_lead_endpoint(request: LeadRequest):
    """
    Receive incoming lead, qualify with Claude, store in Supabase, send SMS.
    This is the core Phase 1 demo endpoint.
    """
    try:
        # 1. Create lead record
        lead_id = create_lead(
            customer_name=request.customer_name,
            phone=request.phone,
            message=request.message,
            service_type=request.service_type
        )
        
        # 2. Qualify with Claude
        qualification = qualify_lead(
            customer_name=request.customer_name,
            phone=request.phone,
            message=request.message,
            service_type=request.service_type
        )
        
        # 3. Update lead status
        update_lead_status(lead_id, "qualified")
        
        # 4. Log qualification action
        log_action(
            lead_id=lead_id,
            action_type="qualification",
            description=f"Urgency: {qualification['urgency_score']}/10, Confidence: {qualification['confidence']}",
            agent_name="claude_lead_qualifier"
        )
        
        # 5. Send SMS if confidence >85%
        sms_sent = False
        if qualification["confidence"] > 0.85:
            try:
                send_sms(request.phone, qualification["response_text"])
                log_action(lead_id, "sms_sent", f"Response: {qualification['response_text']}", "twilio")
                sms_sent = True
            except Exception as e:
                log_action(lead_id, "sms_failed", str(e), "twilio")
        else:
            log_action(lead_id, "escalated", "Low confidence - requires human review", "orchestrator")
        
        return LeadResponse(
            lead_id=lead_id,
            urgency_score=qualification["urgency_score"],
            action=qualification["action"],
            response_text=qualification["response_text"],
            sms_sent=sms_sent
        )
    
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@router.get("/actions")
async def get_actions(limit: int = 50):
    """Get recent actions for dashboard."""
    from services.airtable_service import get_all_actions
    return get_all_actions(limit)
```

```typescript
'use client';

import { useEffect, useState } from 'react';

interface Action {
  id: string;
  fields: {
    lead_id?: string[];
    customer_name?: string;
    phone?: string;
    action_type?: string;
    description?: string;
    agent_name?: string;
    timestamp?: string;
  };
}

export default function MissionControl() {
  const [actions, setActions] = useState<Action[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const fetchActions = async () => {
      try {
        const response = await fetch('http://localhost:8000/leads/actions', {
          headers: { 'Content-Type': 'application/json' }
        });
        const data = await response.json();
        setActions(data);
        setLoading(false);
      } catch (error) {
        console.error('Error fetching actions:', error);
        setLoading(false);
      }
    };

    // Fetch immediately
    fetchActions();

    // Poll every 10 seconds
    const interval = setInterval(fetchActions, 10000);
    return () => clearInterval(interval);
  }, []);

  const getActionColor = (actionType: string) => {
    const colors: Record<string, string> = {
      'qualification': 'bg-blue-100 text-blue-800',
      'sms_sent': 'bg-green-100 text-green-800',
      'escalated': 'bg-orange-100 text-orange-800',
      'sms_failed': 'bg-red-100 text-red-800',
    };
    return colors[actionType] || 'bg-gray-100 text-gray-800';
  };

  const getAgentIcon = (agentName: string) => {
    const icons: Record<string, string> = {
      'claude_lead_qualifier': '🧠',
      'twilio': '📱',
      'orchestrator': '🎯',
      'system': '⚙️',
    };
    return icons[agentName] || '🤖';
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-900 to-slate-800 p-8">
      <div className="max-w-4xl mx-auto">
        {/* Header */}
        <div className="mb-8">
          <h1 className="text-4xl font-bold text-white mb-2">Mission Control</h1>
          <p className="text-slate-400">HVAC AI Operations Dashboard — Phase 1 Demo</p>
        </div>

        {/* Status Bar */}
        <div className="grid grid-cols-3 gap-4 mb-8">
          <div className="bg-slate-700 rounded-lg p-4 text-white">
            <div className="text-sm text-slate-400">Total Actions</div>
            <div className="text-3xl font-bold">{actions.length}</div>
          </div>
          <div className="bg-green-900 rounded-lg p-4 text-green-100">
            <div className="text-sm">SMS Sent</div>
            <div className="text-3xl font-bold">
              {actions.filter(a => a.fields.action_type === 'sms_sent').length}
            </div>
          </div>
          <div className="bg-orange-900 rounded-lg p-4 text-orange-100">
            <div className="text-sm">Escalated</div>
            <div className="text-3xl font-bold">
              {actions.filter(a => a.fields.action_type === 'escalated').length}
            </div>
          </div>
        </div>

        {/* Action Feed */}
        <div className="space-y-3">
          <h2 className="text-lg font-semibold text-white mb-4">Live Action Feed</h2>
          {loading ? (
            <div className="text-slate-400">Loading actions...</div>
          ) : actions.length === 0 ? (
            <div className="bg-slate-700 rounded-lg p-6 text-slate-400 text-center">
              No actions yet. Send a test lead to /leads/qualify to see activity.
            </div>
          ) : (
            actions.map((action) => (
              <div
                key={action.id}
                className="bg-slate-700 rounded-lg p-4 border-l-4 border-blue-500 hover:bg-slate-600 transition"
              >
                <div className="flex items-start justify-between mb-2">
                  <div className="flex items-center gap-3">
                    <span className="text-2xl">
                      {getAgentIcon(action.fields.agent_name || '')}
                    </span>
                    <div>
                      <div className="text-white font-semibold">
                        {action.fields.action_type?.replace('_', ' ').toUpperCase()}
                      </div>
                      <div className="text-sm text-slate-400">
                        {action.fields.agent_name || 'unknown'}
                      </div>
                    </div>
                  </div>
                  <span
                    className={`px-3 py-1 rounded-full text-sm font-medium ${getActionColor(
                      action.fields.action_type || ''
                    )}`}
                  >
                    {action.fields.action_type}
                  </span>
                </div>
                <div className="text-slate-300 text-sm mb-2">{action.fields.description}</div>
                <div className="text-xs text-slate-500">
                  {action.fields.timestamp &&
                    new Date(action.fields.timestamp).toLocaleString()}
                </div>
              </div>
            ))
          )}
        </div>
      </div>
    </div>
  );
}
```
